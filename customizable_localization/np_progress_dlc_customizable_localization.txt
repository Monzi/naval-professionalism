defined_text = { # Icon Tooltip
	name = naval_professionalism_monthly_icon_cl
	text = {
		localisation_key = string_np_monthly_on_icon
		trigger = {
			OR = {
				AND = { # If country does not have VOC Indiamen then give penalty for being over 75%
					light_ship_fraction = 0.75
					if = {
						limit = { has_dlc = "Domination" }
						NOT = { has_reform = stadthalter_monarchy_reform }
						NOT = { has_reform = dutch_republic }
						NOT = { has_reform = dutch_theocracy_reform }
						NOT = {
							any_owned_province = {
								has_province_modifier = ned_fort_of_batavia
							}
						}
					}
				}
				AND = { # If country does have VOC Indiamen then give penalty for being under 25%
					NOT = { light_ship_fraction = 0.25 }
					if = {
						limit = { has_dlc = "Domination" }
						OR = {
							has_reform = stadthalter_monarchy_reform # Dutch Government Reform
							has_reform = dutch_republic # Dutch Government Reform
							has_reform = dutch_theocracy_reform # Dutch Government Reform
							any_owned_province = { # Dutch Mission
								has_province_modifier = ned_fort_of_batavia
							}
						}
					}
				}
			}
		}
	}
}

country_event = { # Initialize Naval Professionalism
	id = naval_professionalism_events.0001
	hidden = yes
	trigger = { # For initializing Naval Professionalism for countries created after game start
		NOT = { has_country_flag = has_naval_professionalism }
	}
	
	immediate = {
		set_variable = {
			which = navalProfessionalismVariable
			value = 0
		}
		set_variable = {
			which = navalProfessionalismControlVariable
			value = 0
		}
		set_country_flag = has_naval_professionalism
	}

	option = {
		name = naval_professionalism_events.0001.a
	}
}

is_at_min_np = {
	check_variable = {
		which = navalProfessionalismVariable
		value = 0
	}
	NOT = {
		check_variable = {
			which = navalProfessionalismVariable
			value = 1
		}
	}
}

is_less_than_5 = {
	check_variable = {
		which = navalProfessionalismVariable
		value = 0
	}
	NOT = {
		check_variable = {
			which = navalProfessionalismVariable
			value = 5
		}
	}
}

is_at_max_np = {
	check_variable = {
		which = navalProfessionalismVariable
		value = 100
	}
	NOT = {
		check_variable = {
			which = navalProfessionalismVariable
			value = 101
		}
	}
}

is_under_np_limit = { # Checks if navalProfessionalismVariable is under 0
	NOT = {
		check_variable = {
			which = navalProfessionalismVariable
			value = 0
		}
	}
	check_variable = {
		which = navalProfessionalismVariable
		value = -100
	}
}

is_over_np_limit = { # Checks if navalProfessionalismVariable is over 100
	check_variable = {
		which = navalProfessionalismVariable
		value = 101
	}
}

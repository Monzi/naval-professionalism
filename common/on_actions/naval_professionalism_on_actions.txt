# country
on_startup = {
	events = {
		naval_professionalism_events.0001 # Initialize Naval Professionalism
	}
}

# country
on_monthly_pulse = {
	if = {
		limit = { has_country_flag = np_is_studying }
		add_tiny_np_progress = yes
	}
	events = {
		naval_professionalism_events.0002 # Sets / Updates modifiers based on level of Naval Professionalism
		naval_professionalism_events.0003 # Monthly check if country has over/under light ship limit
	}
}

# country
on_admiral_recruited = {
	add_small_np_progress = yes
}

# ROOT = Tag that captured the flagship, FROM = Tag that lost the flagship
on_flagship_captured = {
	if = {
		limit = { has_dlc = "Golden Century" }
		FROM = {
			subtract_medium_np_progress = yes
		}
		ROOT = {
			add_medium_np_progress = yes
		}
	}
}

# ROOT = Tag that destroyed the flagship, FROM = Tag that lost the flagship
on_flagship_destroyed = {
	if = {
		limit = { has_dlc = "Golden Century" }
		FROM = {
			subtract_medium_np_progress = yes
		}
	}
}
